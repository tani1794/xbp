score = 0

for i, q in enumerate(quiz_data, start=1):
    print(f"第{i}問")
    for f in q["features"]:
        print("・" + f)
    print("→ この人物は誰？\n")

    answer = input("あなたの答え: ").strip()

    if answer == q["answer"]:
        print("✅ 正解です！🎉\n")
        score += 1
    else:
        print(f"❌ 残念！正解は『{q['answer']}』でした。\n")
